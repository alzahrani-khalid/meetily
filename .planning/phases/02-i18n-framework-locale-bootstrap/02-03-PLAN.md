```typescript
'use client';
import { NextIntlClientProvider } from 'next-intl';
import type { ReactNode } from 'react';

export type Locale = 'en' | 'ar';
export type Messages = Record<string, unknown>;  // opaque to wrapper; next-intl validates at runtime

interface I18nProviderProps {
  locale: Locale;
  messages: Messages;
  children: ReactNode;
}

export function I18nProvider({ locale, messages, children }: I18nProviderProps) {
  return (
    <NextIntlClientProvider locale={locale} messages={messages}>
      {children}
    </NextIntlClientProvider>
  );
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| npm registry → `next-intl` install | Supply chain — adding a new dependency crosses this boundary once at install time. |
| npm registry → shadcn CLI → generated component files | Supply chain — shadcn CLI fetches from its registry and writes files locally. |
| Static JSON message files | No trust boundary — developer-authored content with no runtime user input. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-12 | Supply chain (Tampering) | `next-intl@^3.26.x` | mitigate | Pin to 3.26.x (no major version drift). next-intl is a widely used, audited package. pnpm-lock.yaml captures the exact resolved version + integrity hash. Acceptable standard dev dep trust. |
| T-02-13 | Supply chain (Tampering) | `shadcn@latest` CLI | mitigate | Runs once at install time via `pnpm dlx`. Generates static TSX files that the reviewer can inspect. Post-generation, the CLI is not invoked again. Files are committed to the repo and reviewed in the same commit. |
| T-02-14 | XSS via message catalogue | `messages/en.json`, `messages/ar.json` | accept | Both files ship developer-authored static content. next-intl sanitizes interpolations by default — the only interpolation placeholder is `{lang}`, which resolves to a value from the same catalogue (not user input). No dynamic HTML. No user input enters the catalogues. |
| T-02-15 | Information disclosure | Message catalogue | accept | Catalogues contain no secrets, no PII, no API keys. Strings are UI copy. |
| T-02-16 | Prototype pollution | JSON.parse of messages | accept | Standard JSON.parse has no prototype-pollution vector (it does not merge into any target). next-intl consumes the parsed object as a typed message map. No `Object.assign` into an untrusted target. |
| T-02-17 | Coupling leak (D-07 invariant breach) | Any file other than I18nProvider importing `from 'next-intl'` | mitigate | Task 1 Step D grep verifies single-file coupling. The acceptance criterion enforces exactly one hit. Plan 04 executor will not need to import `useTranslations` in layout.tsx because Plan 04 only wires the provider; Plan 05 consumers import from `@/providers/I18nProvider` which re-exports `useTranslations` from next-intl (re-export added in Task 1 Step C). |

Plan 05 consumers therefore do `import { useTranslations } from '@/providers/I18nProvider'` — never from `'next-intl'` directly. This keeps D-07 literally true: `I18nProvider.tsx` is the only file in the repo that imports from `next-intl`.

**Severity summary:** No HIGH findings. Supply chain (LOW/mitigate via version pinning), coupling (mitigated by Task 1 Step D grep). Plan unblocked.
